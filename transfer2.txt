# Assuming your pandas DataFrame is called 'df'
# And your existing Excel file is named 'existing_file.xlsx'

# Load the DataFrame
df = pd.DataFrame({
    'Column1': [1, 2, 3],
    'Column2': ['A', 'B', 'C'],
})

# Load the existing Excel file
file_path = 'existing_file.xlsx'
book = load_workbook(file_path)

# Select the first worksheet
sheet = book.worksheets[0]

# Clear the contents of the first worksheet
for row in sheet:
    for cell in row:
        cell.value = None

# Get the dimensions of the DataFrame
rows, cols = df.shape

# Write the DataFrame to the first worksheet, starting from cell A1
for r in dataframe_to_rows(df, index=False, header=True):
    sheet.append(r)

# Save the modified Excel file
book.save(file_path)
