Sub MergeColumnsTwoAtATime()
    Dim ws As Worksheet
    Dim lastColumn As Long
    Dim col As Long
    
    ' Change "Sheet1" to the name of your worksheet
    Set ws = ThisWorkbook.Worksheets("Sheet1")
    
    ' Find the last column in row 3
    lastColumn = ws.Cells(3, ws.Columns.Count).End(xlToLeft).Column
    
    ' Loop through the columns two at a time and merge them
    For col = 1 To lastColumn Step 2
        ws.Range(ws.Cells(3, col), ws.Cells(3, col + 1)).Merge
    Next col
End Sub
