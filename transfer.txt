Sub PasteApiResponse()
    Dim responseText As String
    responseText = 'Your API response.text here'

    Dim dataArray() As String
    dataArray = Split(responseText, vbCr) ' Split rows based on vbCR delimiter

    Dim rowIndex As Long
    Dim colIndex As Long
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Sheet1") ' Change "Sheet1" to your target sheet name

    ' Loop through the rows and columns and paste the data
    For rowIndex = LBound(dataArray) To UBound(dataArray)
        Dim rowItems() As String
        rowItems = Split(dataArray(rowIndex), ",") ' Split row items based on comma delimiter

        For colIndex = LBound(rowItems) To UBound(rowItems)
            ws.Cells(rowIndex + 1, colIndex + 1).Value = rowItems(colIndex)
        Next colIndex
    Next rowIndex
End Sub
