in vba, i want to copy a worksheet called "HistoryData" from a workbook called holder.xls in a given file path. I want to paste the HistoryData sheet into the current workbook, overriding or deleting the HistoryData sheet that already exists. That is, current workbook has a sheet called "HistoryData". I want to paste and replace that sheet with "HistoryData" sheet from holder.xls
Sub CopySheetFromOtherWorkbook()
    Dim sourcePath As String
    Dim sourceWorkbook As Workbook
    Dim targetSheet As Worksheet
    
    ' Set the source file path
    sourcePath = "C:\YourFolderPath\holder.xls"
    
    ' Open the source workbook
    Set sourceWorkbook = Workbooks.Open(sourcePath)
    
    ' Check if "HistoryData" sheet exists in the current workbook
    On Error Resume Next
    Set targetSheet = ThisWorkbook.Worksheets("HistoryData")
    On Error GoTo 0
    
    ' If "HistoryData" sheet exists, delete it
    If Not targetSheet Is Nothing Then
        Application.DisplayAlerts = False
        targetSheet.Delete
        Application.DisplayAlerts = True
    End If
    
    ' Copy "HistoryData" sheet from the source workbook to the current workbook
    sourceWorkbook.Sheets("HistoryData").Copy Before:=ThisWorkbook.Sheets(1)
    
    ' Close the source workbook without saving changes
    sourceWorkbook.Close False
End Sub
