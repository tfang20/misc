import openpyxl

def add_right_and_bottom_borders(file_path, sheet_name, start_row):
    workbook = openpyxl.load_workbook(file_path)
    sheet = workbook[sheet_name]

    for row_num in range(start_row, sheet.max_row + 1):
        for col_num in range(1, sheet.max_column + 1):
            cell = sheet.cell(row=row_num, column=col_num)

            # Check if the cell has a right border
            if cell.border.right.style:
                # Add right and bottom borders to the cell
                cell.border = openpyxl.styles.Border(
                    left=cell.border.left,
                    right=cell.border.right,
                    top=cell.border.top,
                    bottom=openpyxl.styles.Side(border_style="thin"),
                )

    workbook.save(file_path)

# Example usage:
excel_file_path = "path/to/your/excel_file.xlsx"
sheet_name = "Sheet1"  # Replace with the name of your sheet
start_row = 2  # Start checking from row 2 (skip header row if any)

add_right_and_bottom_borders(excel_file_path, sheet_name, start_row)
