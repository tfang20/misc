Sub FilterDatesByMonths()
    Dim ws As Worksheet
    Dim dateRange As Range
    Dim cell As Range
    Dim filterCriteria As String
    
    ' Set the worksheet reference
    Set ws = ThisWorkbook.Worksheets("Sheet1") ' Change to your worksheet name
    
    ' Set the range containing your dates (adjust the range as needed)
    Set dateRange = ws.Range("A2:A100") ' Change to your date range
    
    ' Clear existing filters
    dateRange.AutoFilter
    
    ' Build the filter criteria for January and February
    filterCriteria = "1,2"
    
    ' Apply the filter
    dateRange.AutoFilter Field:=1, Criteria1:=Split(filterCriteria, ","), Operator:=xlFilterValues
    
    ' Loop through visible cells and do something (replace this with your desired action)
    For Each cell In dateRange.SpecialCells(xlCellTypeVisible)
        ' Do something with the visible cell, e.g., copy to another location
        ' Example: cell.Copy Destination:=ws.Range("B2")
    Next cell
End Sub
